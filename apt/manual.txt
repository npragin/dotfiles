# Manual installation instructions for GUI apps not available via Flatpak

# ============================================================================
# IMPORTANT: CLI Tools Strategy
# ============================================================================
# All CLI tools (bat, eza, fzf, gh, ripgrep, starship, zoxide, etc.) are
# installed via Homebrew on BOTH macOS and Linux using the Brewfile.
# This ensures:
# - Consistent versions across platforms
# - Same binary names (no batcat/fdfind confusion)
# - Unified package management
#
# To install Homebrew on Linux:
# /bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
# ============================================================================

# === Zinit (Zsh plugin manager) ===
# Not available via Homebrew or package managers - requires manual git clone
bash -c "$(curl --fail --show-error --silent --location https://raw.githubusercontent.com/zdharma-continuum/zinit/HEAD/scripts/install.sh)"

# === Miniforge (Conda) ===
wget https://github.com/conda-forge/miniforge/releases/latest/download/Miniforge3-Linux-x86_64.sh
bash Miniforge3-Linux-x86_64.sh -b -p $HOME/miniforge3
rm Miniforge3-Linux-x86_64.sh

# === Fonts ===
# Meslo Nerd Font
mkdir -p ~/.local/share/fonts
cd ~/.local/share/fonts
wget https://github.com/ryanoasis/nerd-fonts/releases/download/v3.1.1/Meslo.zip
unzip Meslo.zip
rm Meslo.zip
fc-cache -fv
